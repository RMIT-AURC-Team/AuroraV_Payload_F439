12uint8_t systemStatus(SPI_HandleTypeDef *hspi1, SPI_HandleTypeDef *hspi2, I2C_HandleTypeDef* hi2c1, I2C_HandleTypeDef* hi2c2) {
13 uint8_t retVal = 0x00;
14
15 // Check BME280_1
16 if(readBME280_id_reg(hi2c1, 1) != 0x60) { // Expected 0x60
17 retVal = retVal | 0x10;
18 }
19
20 // Check BME280_0
21 if(readBME280_id_reg(hi2c1, 0) != 0x60) { // Expected 0x60
22 retVal = retVal | 0x08;
23 }
24
25 // Check Accelerometer
26 if(readAccel_whoami(hi2c2) != 0xE5) { // Expected 0xE5
27 retVal = retVal | 0x04;
28 }
29
30 // Check SPIFlash_1
31 uint8_t manu[2] = {0, 0};
32 read_manufacturer_id(manu, hspi2, cs_spi2);
33 if(manu[0] != 0xEF) { // Expected 0xEF
34 retVal = retVal | 0x02;
35 }
36
37 // Check SPIFlash_0
38 uint8_t manu0[2] = {0, 0};
39 read_manufacturer_id(manu0, hspi1, cs_spi1);
40 if(manu0[0] != 0xEF) { // Expected 0xEF
41 retVal = retVal | 0x01;
42 }
43 return retVal;
44}

61void readFlashToUART(SPI_HandleTypeDef *hspi, UART_HandleTypeDef *huart, GPIO_Config config) {
62 uint32_t num_of_pages = next_blank_page;
63 if(num_of_pages == 0) {
64 num_of_pages = PAGE_SIZE;
65 }
66 num_of_pages = num_of_pages/PAGE_SIZE;
67 uint32_t address = 0;
68 HAL_GPIO_WritePin(GPIOB, GPIO_PIN_7,GPIO_PIN_SET); // Activate the "write out" LED
69
70 for (int i = 0; i < (num_of_pages); i++) {
71 uint8_t page[PAGE_SIZE];
72 read_page_spi(page, hspi, address, config);
73 uart_transmit_page(huart, page); // Transmit the data//
74 address += PAGE_SIZE;
75 }
76
77 HAL_GPIO_WritePin(GPIOB, GPIO_PIN_7,GPIO_PIN_RESET); // Deactivate the "write out" LED
78}

157/***************************************************************************************************************
158 * Process All Sensors
159 */
160void readAllSensors(I2C_HandleTypeDef* hi2c_accel, I2C_HandleTypeDef* hi2c_temp, RTC_HandleTypeDef* hrtc) {
161 // Read Accelerometer Data
162 readAccelerometer(accel_data, hi2c_accel);
163
164 // Read BME280 Data
165 readTempHumPres(bme280_data_0, hi2c_temp, 0);
166 readTempHumPres(bme280_data_1, hi2c_temp, 1);
167
168 uint16_t time = getTimestampMilliseconds(hrtc);
169 uint8_t array_ptr = 0;
170
171 // Store the new read in the buffer if there is space
172 if (byte_tracker < (PAGE_SIZE - READ_SIZE)) {
173 // Store the timestamp in the buffer (Little Endian)
174 data_buffer_tx[buffer_tracker][byte_tracker + array_ptr] = (uint8_t) (time & 0xFF); // Least significant byte (LSB)
175 array_ptr += 1;
176 data_buffer_tx[buffer_tracker][byte_tracker + array_ptr] = (uint8_t) ((time >> 8) & 0xFF); // Most significant byte (MSB)
177 array_ptr += 1;
178
179 // Store the accelerometer data
180 for (int i = 0; i < 6; i++) {
181 data_buffer_tx[buffer_tracker][byte_tracker + array_ptr] = accel_data[i];
182 array_ptr += 1;
183 }
184
185 // Store the BME280_0 Data
186 for (int i = 0; i < 6; i++) {
187 data_buffer_tx[buffer_tracker][byte_tracker + array_ptr] = bme280_data_0[i];
188 array_ptr += 1;
189 }
190
191 // Store the BME280_1 Data
192 for (int i = 0; i < 6; i++) {
193 data_buffer_tx[buffer_tracker][byte_tracker + array_ptr] = bme280_data_1[i];
194 array_ptr += 1;
195 }
196
197 // Store the flight state and peripheral status in the final byte
198 data_buffer_tx[buffer_tracker][byte_tracker + array_ptr] = combine_system_status();
199 array_ptr += 1;
200
201 byte_tracker = byte_tracker + (array_ptr);
202 }
203}
