851void handleUART() {
852 UART_HandleTypeDef *huart = &huart2;
853
854 // Send Heartbeat to UART (data_rx[0] = "h")
855 if (UARTRxData[0] == 0x68) {
856 heartbeatUART(huart);
857 send_uart_hex(huart, sysStatus);
858 }
859
860/***************************** SPI Flash ************************************************/
861 // Erase specified flash chip (data_rx[0] = "e")
862 else if (UARTRxData[0] == 0x65) {
863 HAL_GPIO_WritePin(led_green.GPIOx, led_green.GPIO_Pin, GPIO_PIN_SET); // Activate the "write out" LED
864 if(decodeASCII(UARTRxData[1]) == 0) {
865 eraseFlashSPI(&hspi1, huart, cs_spi1);
866 } else if (decodeASCII(UARTRxData[1]) == 1) {
867 eraseFlashSPI(&hspi2, huart, cs_spi2);
868 }
869 int next_blank_page0 = find_next_blank_page(&hspi1, &end_of_flash, cs_spi1);
870 int next_blank_page1 = find_next_blank_page(&hspi2, &end_of_flash, cs_spi2);
871
872 // Assign the value of next_blank_page to the larger of next_blank_page0 and next_blank_page1
873 next_blank_page = (next_blank_page0 > next_blank_page1) ? next_blank_page0 : next_blank_page1;
874 HAL_GPIO_WritePin(led_green.GPIOx, led_green.GPIO_Pin,GPIO_PIN_RESET); // Deactivate the "write out" LED
875 }
876
877 // Read data from specified flash chip (data_rx[0] = "r")
878 else if (UARTRxData[0] == 0x72) {
879 if(decodeASCII(UARTRxData[1]) == 0) {
880 readFlashToUART(&hspi1, huart, cs_spi1);
881 } else if (decodeASCII(UARTRxData[1]) == 1) {
882 readFlashToUART(&hspi2, huart, cs_spi2);
883 }
884 }
885
886 // Read Manufacturer over SPI (data_rx[0] = "m")
887 else if (UARTRxData[0] == 0x6d) {
888 if(decodeASCII(UARTRxData[1]) == 0) {
889 readFlashManuSPI(&hspi1, huart, cs_spi1);
890 } else if (decodeASCII(UARTRxData[1]) == 1) {
891 readFlashManuSPI(&hspi2, huart, cs_spi2);
892 }
893 }
894
895 // Write a page over SPI (data_rx[0] = "w")
896 else if (UARTRxData[0] == 0x77) {
897 if(decodeASCII(UARTRxData[1]) == 0) {
898 writePageSPI_W(&hspi1, huart, cs_spi1);
899 } else if (decodeASCII(UARTRxData[1]) == 1) {
900 writePageSPI_W(&hspi2, huart, cs_spi2);
901 }
902 }
903
904 // Software reset flash chip over SPI (data_rx[0] = "x")
905 else if (UARTRxData[0] == 0x78) {
906 if(decodeASCII(UARTRxData[1]) == 0) {
907 resetSPIFlash(&hspi1, huart, cs_spi1);
908 } else if (decodeASCII(UARTRxData[1]) == 1) {
909 resetSPIFlash(&hspi2, huart, cs_spi2);
910 }
911 }
912
913/*********************************** I2C Accelerometer ***********************************/
914 // Read Accelerometer WhoAmI (data_rx[0] = "c")
915 else if (UARTRxData[0] == 0x63) {
916 checkAccelWhoAmI(i2c_accel, huart);
917 }
918
919 // Read the accelerometer and print to the UART[0] (data_rx [0] = "a")
920 else if (UARTRxData[0] == 0x61) {
921 accelToUART(huart);
922 }
923
924/********************************** I2C BME280 *******************************************/
925 // Read the temp sensor ID and print to the UART[0] (data_rx [0]= "b")
926 else if (UARTRxData[0] == 0x62) {
927 readTempSensorID(i2c_bme280, huart, decodeASCII(UARTRxData[1]));
928 }
929
930 // Read the temp sensor calibration registers and print to the UART (data_rx[0] = "p")
931 else if (UARTRxData[0] == 0x70) {
932 readTempCalibration(i2c_bme280, huart, decodeASCII(UARTRxData[1]));
933 }
934
935 // Read the temp sensor and print to the UART[0] (data_rx [0]= "t")
936 else if (UARTRxData[0] == 0x74) {
937 readTempSensor(huart, decodeASCII(UARTRxData[1]));
938 }
939
940 /********************************** CAN Bus *******************************************/
941 // Send the second byte received over the CAN bus as a payload with id 0x700 (data_rx [0]= "n")
942 else if (UARTRxData[0] == 0x6E) {
943 uint8_t TxData[1] = {UARTRxData[1]};
944 sendCAN_TxMessage(&hcan2, 1, TxData, &CAN_TxMailbox, DUMMY_ID);
945 }
946
947 UARTRxData[0] = 0x00;
948 UARTRxData[1] = 0x00;
949}

959uint8_t combine_system_status() {
960 // Mask status to use only bits 4:0
961 uint8_t masked_status = sysStatus & 0x1F; // 0001 1111b
962
963 // Mask flight_state to use only bits 2:0
964 uint8_t masked_flight_state = flight_state & 0x07; // 0000 0111b
965 // Shift flight_state to the correct position (bits 7:5)
966 uint8_t shifted_flight_state = masked_flight_state << 5;
967
968 // Combine the masked_status and shifted_flight_state
969 uint8_t combined_value = masked_status | shifted_flight_state;
970
971 return combined_value;
972}
