21uint8_t check_busy(SPI_HandleTypeDef *hspi, GPIO_Config config, int timeout) {
22 uint8_t ret_val = 0x00;
23
24 // Get the current time
25 uint32_t startTick = HAL_GetTick();
26
27 // Read the status register to ensure no write is currently in progress
28 uint8_t busy = 0x01;
29 while(busy) {
30 // Check if there is a write in progress
31 busy = (check_status_register(hspi, config) & 0x01);
32
33 uint32_t currTick = HAL_GetTick();
34
35 // Check if the timeout has been reached
36 if ((currTick - startTick) >= timeout) {
37 ret_val = 0x01;
38 break;
39 }
40 }
41
42 return ret_val;
43}
